package com.quiz.evaluation;

import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;
import org.springframework.web.bind.annotation.*;
import java.util.*;

@SpringBootApplication
@RestController
@RequestMapping("/api/quiz")
@CrossOrigin(origins = "*")
public class QuizBackendApplication {

    public static void main(String[] args) {
        SpringApplication.run(QuizBackendApplication.class, args);
    }

    private static final List<Question> quizDatabase = Arrays.asList(
        new Question(1, "OOPS", "Which OOPS concept provides 'Security' to data?", Arrays.asList("Inheritance", "Encapsulation", "Polymorphism", "Abstraction"), 1),
        new Question(2, "OOPS", "Which of the following is used to achieve 100% Abstraction?", Arrays.asList("Abstract Class", "Interface", "Encapsulation", "Static Method"), 1),
        new Question(3, "JAVA", "Which class is the superclass of every class in Java?", Arrays.asList("String", "Main", "Object", "System"), 2),
        new Question(4, "JAVA", "What is the size of 'char' in Java?", Arrays.asList("4 bits", "8 bits", "16 bits", "32 bits"), 2)
    );

    @GetMapping("/questions")
    public List<Map<String, Object>> getQuestions(@RequestParam(required = false) String category) {
        List<Map<String, Object>> filtered = new ArrayList<>();
        for (Question q : quizDatabase) {
            if (category == null || q.category.equalsIgnoreCase(category)) {
                Map<String, Object> map = new HashMap<>();
                map.put("id", q.id);
                map.put("category", q.category);
                map.put("text", q.text);
                map.put("options", q.options);
                filtered.add(map);
            }
        }
        return filtered;
    }

    @PostMapping("/evaluate")
    public Map<String, Object> evaluate(@RequestBody Map<Integer, Integer> answers) {
        int score = 0;
        int total = 0;
        for (Question q : quizDatabase) {
            if (answers.containsKey(q.id)) {
                total++;
                if (answers.get(q.id) == q.correctIndex) score++;
            }
        }
        
        Map<String, Object> res = new HashMap<>();
        double pct = (total == 0) ? 0 : ((double) score / total) * 100;
        res.put("score", score);
        res.put("total", total);
        res.put("percentage", pct);
        res.put("status", pct >= 50 ? "PASSED" : "FAILED");
        return res;
    }

    static class Question {
        int id; String category, text; List<String> options; int correctIndex;
        public Question(int id, String category, String text, List<String> options, int correctIndex) {
            this.id = id; this.category = category; this.text = text; this.options = options; this.correctIndex = correctIndex;
        }
    }
}
